def self.create_participant_added(company_key, acquisition_id, destination_url)
  create company_key, {
    event_type: "ack_participant_added",
    ack_participant_added: {
      campaign_id: acquisition_id
    },
    destination: {
      url: destination_url,
      method: "POST",
      content_type: "application/json"
    }
  }
